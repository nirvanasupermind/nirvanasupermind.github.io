1. us(a, b, c, d) = us(a, b, c, d)[a]
  2. us(a, b, 0, 0)[x] = bx
  3. us(a, b, 1, d)[x] = a
  4. c > 0: us(a, b, c, 0)[x] = us(a, us(a, b, c,0)[x - 1], c - 1, 0)
  5. d > 0: us(a, b, 0, d)[x] = us(a, b, x, d - 1)
  6. c > 0, d > 0: us(a, b, c, d)[0] = us(a, b, 0, d)
  7. c > 0, d > 0, x > 0: us(a, b, c, d)[x] = us(a, us(a, b - 1, c, d)[x], c, d)[x - 1]
